\begin{frame}[fragile]{More Code}
  \begin{verbatim}
def rolldie(N):
    for i in range(1,7):
        if N + i <= 100:
            a[N, N + i] = 1/6
        else:
            a[N, N] += 1/6
  \end{verbatim}
\end{frame}

\begin{frame}[fragile]{Still More Code}

for k in range(100):
    if k not in SaLdic:
        rolldie(k)

for p, q in SaLdic.iteritems():
    a[p, q] = 1 
  \end{verbatim}
